  <script>

    function connectStrava() {

    function toggleCustomEvent(value) {

      const customFields = document.getElementById("customEventFields");

      customFields.style.display = value === "custom" ? "block" : "none";

    }

    function calculateOffset(e) {

      e.preventDefault();

      const eventDropdown = document.getElementById("eventSelect");

      let selectedValue = parseFloat(eventDropdown.value);

      if (eventDropdown.value === "custom") {

        selectedValue = 24.4; // Default estimate for custom events for now

      }

      if (!selectedValue || isNaN(selectedValue)) return;

      const kgCO2 = selectedValue;

      const cost = (kgCO2 / 16).toFixed(2);

      const trees = Math.floor(kgCO2 / 16);

      const plastics = trees * 2;

      const resultBox = document.getElementById("offset-result");

      resultBox.style.display = "block";

      resultBox.innerHTML = `

        <h3>Your Offset Details</h3>

        <p><strong>CO₂ to offset:</strong> ${kgCO2} kg</p>

        <p><strong>Total Cost:</strong> £${cost}</p>

        <p>This will counter <strong>${kgCO2} kg CO₂</strong>, plant <strong>${trees} tree(s)</strong> and collect <strong>${plastics} ocean-bound plastic bottles</strong>.</p>

        <button class="button" onclick="alert('Offset payment system coming soon!')">Confirm and Offset</button>

      `;

    }

  </script>
